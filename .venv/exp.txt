def generate_human_psychological_characters(disorders_type):
    llm=GoogleGenerativeAI(model="gemini-pro", temperature=0.7)

    prompt_template_name= PromptTemplate(
    input_variables=['disorders_type'],
    template=("I have a {disorders_type} and I want to learn about {disorders_type}. Can you give me detailed information and how to care myself")
)
    name_chain = LLMChain(llm=llm, prompt=prompt_template_name)
    response =name_chain({'disorders_type: disorders_type'})
    return response
